    <script>
        const form = document.getElementById('hoursForm');
        const recordsList = document.getElementById('recordsList');
        const totalHoursEl = document.getElementById('totalHours');
        const totalHoursValue = document.getElementById('totalHoursValue');
        const emailSection = document.getElementById('emailSection');
        const sendEmailBtn = document.getElementById('sendEmailBtn');
        const successMessage = document.getElementById('successMessage');
        const dateInput = document.getElementById('date');

        // Establecer fecha de hoy por defecto
        dateInput.valueAsDate = new Date();

        // Cargar registros desde localStorage
        let records = JSON.parse(localStorage.getItem('extraHours')) || [];

        function saveRecords() {
            localStorage.setItem('extraHours', JSON.stringify(records));
        }

        function renderRecords() {
            if (records.length === 0) {
                recordsList.innerHTML = '<div class="empty-state">No hay registros aún. ¡Registra las primeras horas extras!</div>';
                totalHoursEl.style.display = 'none';
                emailSection.style.display = 'none';
                return;
            }

            recordsList.innerHTML = records.map((record, index) => `
                <div class="record-item">
                    <div class="record-header">
                        <div class="record-name">${record.workerName}</div>
                        <div class="record-date">${new Date(record.date).toLocaleDateString('es-ES')}</div>
                    </div>
                    <div class="record-details">
                        <strong>ID:</strong> ${record.workerId || 'N/A'}<br>
                        <strong>Horas:</strong> ${record.hours} horas<br>
                        <strong>Descripción:</strong> ${record.description || 'Sin descripción'}
                    </div>
                </div>
            `).join('');

            // Calcular total de horas
            const total = records.reduce((sum, record) => sum + parseFloat(record.hours), 0);
            totalHoursValue.textContent = total.toFixed(1);
            totalHoursEl.style.display = 'block';
            emailSection.style.display = 'block';
        }

        form.addEventListener('submit', function(e) {
            e.preventDefault();
            
            const newRecord = {
                id: Date.now(),
                workerName: document.getElementById('workerName').value,
                workerId: document.getElementById('workerId').value,
                date: document.getElementById('date').value,
                hours: parseFloat(document.getElementById('hours').value),
                description: document.getElementById('description').value
            };

            records.unshift(newRecord);
            saveRecords();
            renderRecords();

            // Limpiar formulario
            form.reset();
            dateInput.valueAsDate = new Date();

            // Feedback visual
            const btn = form.querySelector('.btn');
            const originalText = btn.textContent;
            btn.textContent = '✅ Registrado!';
            btn.style.background = '#10b981';
            setTimeout(() => {
                btn.textContent = originalText;
                btn.style.background = '';
            }, 2000);
        });

        // Función para enviar email
        sendEmailBtn.addEventListener('click', function() {
            const recipientEmail = document.getElementById('recipientEmail').value;
            const senderName = document.getElementById('senderName').value;

            if (!recipientEmail || !senderName) {
                alert('Por favor completa todos los campos del correo');
                return;
            }

            // Generar reporte HTML
            const totalHours = records.reduce((sum, record) => sum + parseFloat(record.hours), 0);
            const reportHTML = `
                <h2>📊 REPORTE DE HORAS EXTRAS</h2>
                <p><strong>Generado por:</strong> ${senderName}</p>
                <p><strong>Fecha del reporte:</strong> ${new Date().toLocaleDateString('es-ES')}</p>
                <hr>
                <h3>📈 RESUMEN</h3>
                <p><strong>Total de registros:</strong> ${records.length}</p>
                <p><strong>Total de horas extras:</strong> <strong style="font-size: 1.5em; color: #10b981;">${totalHours.toFixed(1)} horas</strong></p>
                <hr>
                <h3>📋 DETALLE DE REGISTROS</h3>
                ${records.map(record => `
                    <div style="background: #f8f9fa; padding: 15px; margin: 10px 0; border-radius: 8px; border-left: 4px solid #4facfe;">
                        <h4>${record.workerName}</h4>
                        <p><strong>Fecha:</strong> ${new Date(record.date).toLocaleDateString('es-ES')}</p>
                        <p><strong>ID:</strong> ${record.workerId || 'N/A'}</p>
                        <p><strong>Horas:</strong> ${record.hours} horas</p>
                        ${record.description ? `<p><strong>Descripción:</strong> ${record.description}</p>` : ''}
                    </div>
                `).join('')}
                <hr>
                <p style="color: #666; font-size: 0.9em;">
                    Este reporte fue generado automáticamente desde el sistema de Horas Extras.
                </p>
            `;

            // Datos para el email
            const emailData = {
                to: recipientEmail,
                subject: `📊 Reporte de Horas Extras - ${new Date().toLocaleDateString('es-ES')} (${records.length} registros, ${totalHours.toFixed(1)} horas)`,
                body: reportHTML
            };

            // Abrir cliente de correo del usuario
            const mailtoUrl = `mailto:${emailData.to}?subject=${encodeURIComponent(emailData.subject)}&body=${encodeURIComponent(emailData.body)}`;
            window.open(mailtoUrl, '_blank');

            // Mostrar mensaje de éxito
            successMessage.style.display = 'block';
            successMessage.scrollIntoView({ behavior: 'smooth' });

            // Limpiar campos
            document.getElementById('recipientEmail').value = '';
            document.getElementById('senderName').value = '';

            setTimeout(() => {
                successMessage.style.display = 'none';
            }, 5000);
        });

        // Renderizar al cargar
        renderRecords();
    </script>
